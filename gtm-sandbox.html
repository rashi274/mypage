-- dbt model: marts/growth/fct_gtm_funnel_conversion.sql
WITH events AS (
    SELECT * FROM {{ ref('stg_segment_events') }}
),
enriched_leads AS (
    SELECT * FROM {{ ref('stg_clay_enriched_leads') }}
)
SELECT 
    DATE_TRUNC(e.timestamp, DAY) AS event_date,
    COUNT(DISTINCT CASE WHEN e.event_name = 'Pageview' THEN e.anonymous_id END) AS visitors,
    COUNT(DISTINCT CASE WHEN e.event_name = 'Signup' THEN e.user_id END) AS signups,
    COUNT(DISTINCT el.email) AS enriched_leads,
    COUNT(DISTINCT CASE WHEN e.event_name = 'Upgrade' THEN e.user_id END) AS conversions,
    SAFE_DIVIDE(
        COUNT(DISTINCT CASE WHEN e.event_name = 'Upgrade' THEN e.user_id END),
        COUNT(DISTINCT CASE WHEN e.event_name = 'Signup' THEN e.user_id END)
    ) * 100 AS conversion_rate
FROM events e
LEFT JOIN enriched_leads el ON e.user_id = el.user_id
GROUP BY 1
ORDER BY 1 DESC;
# Python Serverless function: gtm-webhooks/enrich_lead.py
import requests
import json
import os

CLAY_API_KEY = os.environ.get("CLAY_API_KEY")
HUBSPOT_API_KEY = os.environ.get("HUBSPOT_API_KEY")

def handler(request):
    """
    Called by GTM / Segment webhook on signup.
    Forwards email to Clay for enrichment, then updates HubSpot CRM.
    """
    request_json = request.get_json(silent=True)
    if not request_json or "email" not in request_json:
        return {"status": "error", "message": "Email is required"}, 400
        
    email = request_json["email"]
    
    # 1. Trigger Clay.com Lead Enrichment Workflow
    clay_url = "https://api.clay.run/v3/workflows/enrich-lead"
    headers = {"Authorization": f"Bearer {CLAY_API_KEY}", "Content-Type": "application/json"}
    clay_payload = {"email": email}
    
    clay_response = requests.post(clay_url, json=clay_payload, headers=headers)
    enriched_data = clay_response.json()
    
    # 2. Map enriched attributes to HubSpot schema
    hubspot_payload = {
        "properties": {
            "email": email,
            "firstname": enriched_data.get("first_name"),
            "lastname": enriched_data.get("last_name"),
            "jobtitle": enriched_data.get("job_title"),
            "company": enriched_data.get("company_name"),
            "linkedin_profile": enriched_data.get("linkedin_url"),
            "company_size": enriched_data.get("company_size"),
            "lead_score": enriched_data.get("lead_score")
        }
    }
    
    # 3. Write data to HubSpot
    hs_url = "https://api.hubapi.com/crm/v3/objects/contacts"
    hs_headers = {"Authorization": f"Bearer {HUBSPOT_API_KEY}", "Content-Type": "application/json"}
    requests.post(hs_url, json=hubspot_payload, headers=hs_headers)
    
    return {"status": "success", "message": "Lead enriched and CRM synced"}, 200

function() {
  // Extract domain from lead email stored in Data Layer
  var email = google_tag_manager["GTM-XXXX"].macro('dL - User Email');
  if (email && email.indexOf('@') !== -1) {
    var parts = email.split('@');
    if (parts.length === 2) {
      var domain = parts[1].toLowerCase();
      // Exclude generic public mail domains
      var freeDomains = ["gmail.com", "yahoo.com", "outlook.com", "hotmail.com"];
      if (freeDomains.indexOf(domain) === -1) {
        return domain;
      }
    }
  }
  return null;
}
